function bucketSort($max, $arr, $sort = 'ASC')
{
    // 把排序参数转成大写
    $sort = strtoupper($sort);
    // 填充max+1个桶
    $bucket = array_fill(0, $max+1, 0);
    // 计算每个数字出现的次数
    for ($i=0; $i < $max; $i++) { 
        $bucket[$arr[$i]]++;
    }
    $sortArr = [];
    // 从小到大
    if ($sort === 'ASC') {
        for ($i=0; $i <= $max; $i++) {
            for ($k=1; $k <= $bucket[$i]; $k++) {
                $sortArr[] = $i;
            }
        }
    // 从大到小
    } else if ($sort === 'DESC') {
        for ($i=$max; $i >= 0; $i--) {
            for ($k=1; $k <= $bucket[$i]; $k++) {
                $sortArr[] = $i;
            }
        }
    }
    return $sortArr;
}
$arr = [123,12312,4,312,2352,3532,123,1231];
$result = bucketSort(12312, $arr);
var_dump($result);
// 打印的结果为:
array (size=8)
	0 => int 4
	1 => int 123
	2 => int 123
	3 => int 312
	4 => int 1231
	5 => int 2352
	6 => int 3532
	7 => int 12312
